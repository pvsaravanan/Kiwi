# Sentinel Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Sentinel, a CI memory layer that ingests JUnit XML test failures into Cognee's knowledge graph and answers "have we seen this failure before, and what fixed it?" with a Streamlit graph dashboard for the live demo.

**Architecture:** A Python CLI (`ingest.py`) parses JUnit XML, calls `remember()` (Cognee add+cognify) and `recall()` (Cognee search) against a scoped "sentinel" dataset on Cognee Cloud. A separate `lifecycle.py` handles `improve()` and `forget()`. A Streamlit app (`visualize.py`) shows the memory graph and recall results. Seeded with 20 synthetic historical failures so recall has history to find from day one. No separate infrastructure — Cognee Cloud is hosted.

**Tech Stack:** Python 3.11+, cognee SDK (Cognee Cloud), streamlit, pyvis, python-dotenv, pytest, pytest-asyncio, pytest-mock

## Global Constraints

- Python 3.11+
- Cognee Cloud at platform.cognee.ai — no local Cognee server
- All Cognee I/O is async — use `asyncio.run()` at CLI entry points
- Always scope Cognee calls to `DATASET_NAME` — never use the default dataset (prevents context pollution per PRD §12b)
- Never commit `.env` — only `.env.example`
- All unit tests mock Cognee SDK calls; integration tests marked `@pytest.mark.integration` and skipped by default with `-m "not integration"`
- YAGNI: `browser_agent.py` and `pr_comment.py` are out of scope — do not build them
- **Cognee SDK note:** The Cognee SDK API may evolve. Canonical source: `pip show cognee` then `python -c "import cognee; help(cognee)"`. If any method signature below differs from the installed version, match the installed version.

- [ ] **Step 4: Implement parser.py**

```python
from dataclasses import dataclass
from typing import Optional
import xml.etree.ElementTree as ET


@dataclass
class FailureRecord:
    test_name: str
    class_name: str
    error_message: str
    stack_trace: str
    failure_type: str
    file_hint: str
    run_id: Optional[str] = None
    timestamp: Optional[str] = None


def parse_junit_xml(path: str) -> list[FailureRecord]:
    root = ET.parse(path).getroot()
    suites = root.findall("testsuite") if root.tag == "testsuites" else [root]

    results = []
    for suite in suites:
        for tc in suite.findall("testcase"):
            node = tc.find("failure") or tc.find("error")
            if node is None:
                continue
            class_name = tc.get("classname", "")
            results.append(FailureRecord(
                test_name=tc.get("name", ""),
                class_name=class_name,
                error_message=node.get("message", ""),
                stack_trace=(node.text or "").strip(),
                failure_type=node.get("type", ""),
                file_hint=class_name.replace(".", "/"),
            ))
    return results
```

```python
import asyncio
import json
import cognee
from pathlib import Path
from sentinel.config import DATASET_NAME, setup_cognee


def _format_record(r: dict) -> str:
    return (
        f"Historical Test Failure:\n"
        f"Test: {r['test_name']}\n"
        f"File: {r['file']}\n"
        f"Error Type: {r.get('error_type', 'Unknown')}\n"
        f"Error: {r['error']}\n"
        f"Stack Trace:\n{r.get('stack_trace', 'N/A')}\n"
        f"Date: {r['timestamp']}\n"
        f"Resolution: {r['resolution']}\n"
    )


async def load_seed_data(path: str = "sentinel/seed_data.jsonl") -> None:
    setup_cognee()
    records = [
        json.loads(line)
        for line in Path(path).read_text().strip().split("\n")
        if line.strip()
    ]
    combined = "\n\n---\n\n".join(_format_record(r) for r in records)
    await cognee.add(combined, dataset_name=DATASET_NAME)
    await cognee.cognify(datasets=[DATASET_NAME])
    print(f"[SEED] Loaded {len(records)} historical failures into '{DATASET_NAME}'")


if __name__ == "__main__":
    asyncio.run(load_seed_data())
```

```python
import asyncio
import argparse
import json
import cognee
from sentinel.config import DATASET_NAME, setup_cognee
from sentinel.parser import parse_junit_xml, FailureRecord


def _format_for_memory(f: FailureRecord) -> str:
    return (
        f"Test Failure:\n"
        f"Test: {f.test_name}\n"
        f"Class: {f.class_name}\n"
        f"File: {f.file_hint}\n"
        f"Error Type: {f.failure_type}\n"
        f"Error Message: {f.error_message}\n"
        f"Stack Trace:\n{f.stack_trace}\n"
        f"Run ID: {f.run_id or 'unknown'}\n"
    )


async def remember(failure: FailureRecord) -> None:
    setup_cognee()
    await cognee.add(_format_for_memory(failure), dataset_name=DATASET_NAME)
    await cognee.cognify(datasets=[DATASET_NAME])


async def recall(failure: FailureRecord) -> list:
    setup_cognee()
    query = f"failures similar to: {failure.error_message} in {failure.file_hint}"
    return await cognee.search("GRAPH_COMPLETION", query=query, datasets=[DATASET_NAME])


async def run(xml_path: str, run_id: str | None = None) -> list[dict]:
    failures = parse_junit_xml(xml_path)
    results = []

    for failure in failures:
        failure.run_id = run_id
        print(f"\n[SENTINEL] {failure.test_name}")

        prior = await recall(failure)
        if prior:
            print(f"[RECALL] {len(prior)} prior match(es):")
            for r in prior:
                print(f"  {json.dumps(r, default=str)[:200]}")
        else:
            print("[RECALL] New failure — no history found.")

        await remember(failure)
        print(f"[REMEMBER] Stored in '{DATASET_NAME}'")
        results.append({"failure": failure, "recall": prior})

    return results


if __name__ == "__main__":
    ap = argparse.ArgumentParser(description="Sentinel: CI failure memory layer")
    ap.add_argument("xml_path", help="Path to JUnit XML report")
    ap.add_argument("--run-id", default=None, help="CI run identifier")
    args = ap.parse_args()
    asyncio.run(run(args.xml_path, args.run_id))
```

```python
import asyncio
import argparse
import cognee
from sentinel.config import DATASET_NAME, setup_cognee


async def improve(test_name: str, resolution: str) -> None:
    """Strengthen the fix edge for a confirmed resolution in Cognee's graph."""
    setup_cognee()
    text = (
        f"Confirmed Fix:\n"
        f"Test: {test_name}\n"
        f"Resolution: {resolution}\n"
        f"Status: CONFIRMED_RESOLVED\n"
    )
    await cognee.add(text, dataset_name=DATASET_NAME)
    await cognee.cognify(datasets=[DATASET_NAME])
    print(f"[IMPROVE] Strengthened fix edge for {test_name}")


async def forget(test_name: str) -> None:
    """Mark a stable test's failure history as deprecated in Cognee's graph."""
    setup_cognee()
    text = (
        f"Deprecated Failure:\n"
        f"Test: {test_name}\n"
        f"Status: DEPRECATED_STABLE\n"
        f"Note: Stable for N consecutive runs — pruned from active failure memory.\n"
    )
    await cognee.add(text, dataset_name=DATASET_NAME)
    await cognee.cognify(datasets=[DATASET_NAME])
    print(f"[FORGET] Marked {test_name} as deprecated in '{DATASET_NAME}'")


if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    sub = ap.add_subparsers(dest="command", required=True)

    imp = sub.add_parser("improve", help="Confirm a fix for a test failure")
    imp.add_argument("test_name")
    imp.add_argument("resolution")

    frg = sub.add_parser("forget", help="Prune a resolved test from active memory")
    frg.add_argument("test_name")

    args = ap.parse_args()
    if args.command == "improve":
        asyncio.run(improve(args.test_name, args.resolution))
    elif args.command == "forget":
        asyncio.run(forget(args.test_name))
```

import asyncio
import streamlit as st
from pyvis.network import Network
import streamlit.components.v1 as components
from sentinel.parser import FailureRecord
from sentinel.ingest import recall
from sentinel.lifecycle import improve

st.set_page_config(page_title="Sentinel", layout="wide")
st.title("Sentinel — CI Failure Memory")

col_left, col_right = st.columns([1, 2])

with col_left:
    st.header("Recall Query")
    error_msg = st.text_area("Error message:", height=100,
                              placeholder="Paste the error message from a failing test...")
    file_hint = st.text_input("File hint (optional):", placeholder="payments/webhook.py")

    if st.button("Recall", type="primary") and error_msg:
        failure = FailureRecord(
            test_name="_query_",
            class_name="",
            error_message=error_msg,
            stack_trace="",
            failure_type="",
            file_hint=file_hint,
        )
        with st.spinner("Querying Cognee memory..."):
            results = asyncio.run(recall(failure))
        st.session_state["recall_results"] = results

    st.divider()
    st.header("Confirm Fix")
    fix_test = st.text_input("Test name:")
    fix_resolution = st.text_area("Resolution:", height=80)
    if st.button("Confirm (improve)") and fix_test and fix_resolution:
        with st.spinner("Updating graph..."):
            asyncio.run(improve(fix_test, fix_resolution))
        st.success(f"Fix confirmed for {fix_test}")

with col_right:
    results = st.session_state.get("recall_results", [])
    if results:
        st.header(f"Recall Results ({len(results)} match(es))")
        for r in results:
            with st.expander(str(r)[:100]):
                st.json(r)
    else:
        st.header("Recall Results")
        st.info("Enter an error message and click Recall.")

    st.divider()
    st.header("Memory Graph — Seed Clusters")

    net = Network(height="500px", width="100%", bgcolor="#1e1e1e", font_color="white")
    net.set_options('{"physics": {"enabled": true, "stabilization": {"iterations": 100}}}')

    clusters = [
        ("TimeoutError", "#ff6b6b",
         ["test_payment_webhook", "test_checkout_flow", "test_report_generation"]),
        ("AuthError", "#ffd93d",
         ["test_user_login", "test_admin_dashboard"]),
        ("DBError", "#6bcb77",
         ["test_order_processing"]),
        ("AssertionError", "#4d96ff",
         ["test_product_listing", "test_cache_invalidation", "test_api_rate_limit"]),
        ("ConfigError", "#c77dff",
         ["test_email_notification", "test_websocket_connection"]),
    ]
    for error_type, color, tests in clusters:
        net.add_node(error_type, label=error_type, color=color, size=30, shape="diamond")
        for test in tests:
            net.add_node(test, label=test, color="#aaaaaa", size=15)
            net.add_edge(test, error_type, label="FAILED_WITH")

    components.html(net.generate_html(), height=520)
